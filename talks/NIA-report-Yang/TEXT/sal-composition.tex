\vfill
\BULL \STRESS{Asynchronous composition} : \texttt{M1[]M2}
  \LIST{
  \CIRC Initialization: combining the initializations in different modules.
  $$\id{init\_M1} \land \id{init\_M2}$$
  \CIRC Transition: the union of transition definitions.
  $$\id{trans\_M1\_stat1} \lor \cdots \lor \id{trans\_M2\_stat1} \lor \cdots $$
  }

\vfill
\BULL \RED{Synchronous composition} : \texttt{M1||M2}
  \LIST{
  \CIRC Difficulty: applying saturation on 
  $$(\id{trans\_M1\_stat1} \lor \cdots) \land (\id{trans\_M2\_stat1} \lor \cdots) $$
  }
}
