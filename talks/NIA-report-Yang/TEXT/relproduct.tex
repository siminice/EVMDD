\FOIL{Relational product}

Relational product: 
\LIST{
\BULL input: 
    \LIST{
    \CIRC $L$-level MDD $s$ encoding a set of states
    \CIRC $2L$-level MDD $r$ encoding the next-state function
    }
\BULL output:
MDD $t$ encoding the set of states which can be reached from a state in $s$ by applying the next-state function $r$.
}

\FPSCENTERWIDE{mdd-firing}{8cm}

\begin{equation*}
\bset(t) \!=\! \id{RelProd}(s,r) \Longleftrightarrow
\bset(t[j]) \! = \!\! \bigcup_{\forall i \in \sset_l} \!\!
[
\id{RelProd}(s[i],r[i][j])
],
\end{equation*}
